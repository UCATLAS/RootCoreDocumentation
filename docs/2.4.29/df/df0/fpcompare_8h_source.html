00085 // Decide whether we need to use volatile or not.
00086 #if defined(__FLT_EVAL_METHOD__) && \
00087   (__FLT_EVAL_METHOD__ == 2 || __FLT_EVAL_METHOD__ < 0)
00088   // __FLT_EVAL_METHOD__ < 0 means unspecified.
00089   // Be pessimistic in that case.
00090 # define CXXUTILS_FPCOMPARE_VOLATILE volatile
00091 #elif defined(__i386__) && !defined(__SSE2__)
00092   // On x86, gcc -msse -mfpmath=sse is observed to _not_ generate
00093   // sse fp instructions, but does set __FLT_EVAL_METHOD__ to 0.
00094   // -msse2 -mfpmath=sse does seem to work as expected.
00095   // Special-case this for now; should follow up with a gcc bug report
00096   // if this still happens in current releases.
00097 # define CXXUTILS_FPCOMPARE_VOLATILE volatile
00098 #else
00099 # define CXXUTILS_FPCOMPARE_VOLATILE
00100 #endif
