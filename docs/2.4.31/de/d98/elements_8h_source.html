00058    // UnknownElement - provides a typesafe surrogate for any of the JSON-
00059    //  sanctioned element types. This class allows the Array and Object
00060    //  class to effectively contain a heterogeneous set of child elements.
00061    // The cast operators provide convenient implicit downcasting, while
00062    //  preserving dynamic type safety by throwing an exception during a
00063    //  a bad cast. 
00064    // The object & array element index operators (operators [std::string]
00065    //  and [size_t]) provide convenient, quick access to child elements.
00066    //  They are a logical extension of the cast operators. These child
00067    //  element accesses can be chained together, allowing the following
00068    //  (when document structure is well-known):
00069    //  String str = objInvoices[1]["Customer"]["Company"];
00070    class UnknownElement {
00071 
00072    public:
00073       UnknownElement();
00074       UnknownElement( const UnknownElement& unknown );
00075       UnknownElement( const Object& object );
00076       UnknownElement( const Array& array );
00077       UnknownElement( const Number& number );
00078       UnknownElement( const Boolean& boolean );
00079       UnknownElement( const String& string );
00080       UnknownElement( const Null& null );
00081 
00082       ~UnknownElement();
00083 
00084       UnknownElement& operator = ( const UnknownElement& unknown );
00085 
00086       // implicit cast to actual element type. throws on failure
00087       operator const Object& () const;
00088       operator const Array& () const;
00089       operator const Number& () const;
00090       operator const Boolean& () const;
00091       operator const String& () const;
00092       operator const Null& () const;
00093 
00094       // implicit cast to actual element type. *converts* on failure, and always returns success
00095       operator Object& ();
00096       operator Array& ();
00097       operator Number& ();
00098       operator Boolean& ();
00099       operator String& ();
00100       operator Null& ();
00101 
00102       // provides quick access to children when real element type is object
00103       UnknownElement& operator[] ( const std::string& key );
00104       const UnknownElement& operator[] ( const std::string& key ) const;
00105 
00106       // provides quick access to children when real element type is array
00107       UnknownElement& operator[] ( size_t index );
00108       const UnknownElement& operator[] ( size_t index ) const;
00109 
00110       // implements visitor pattern
00111       void Accept( ConstVisitor& visitor ) const;
00112       void Accept( Visitor& visitor );
00113 
00114       // tests equality. first checks type, then value if possible
00115       bool operator == ( const UnknownElement& element ) const;
