273  inline float compute_calibHitsShowerDepth(const std::array<float, 4>& cl, float eta)
274  {
275  const std::array<float, 4> radius(get_MVAradius(eta));
276 
277  // loop unrolling
278  const float denominator = cl[0] + cl[1] + cl[2] + cl[3];
279  if (denominator == 0) return 0.;
280 
281  return (radius[0] * cl[0]
282  + radius[1] * cl[1]
283  + radius[2] * cl[2]
284  + radius[3] * cl[3]) / denominator;
