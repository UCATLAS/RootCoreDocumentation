00032 // ******************************************************************************
00033 // class MuonIdHelper
00034 // ******************************************************************************
00035 //
00036 // Description
00037 // -----------
00038 // This is a base class for the Muon Identifier and Range factory classes.
00039 // ATLAS note ATL-MUON-2001-014 provides a complete description of the
00040 // hierarchical identifier scheme.  MuonIdHelper provides an interface to the
00041 // common station "prefix" of the identifier, as described below.
00042 //
00043 // Field           Value               Notes
00044 // ==============================================================================
00045 // StationName     [0..n]              maps to "BIL", "EMS", "T1F", etc.
00046 //                                     there can be holes in the list of indices
00047 // StationEta      integer             increases with eta (0 at eta=0)
00048 // StationPhi      non-zero integer    increases with phi
00049 // Technology      [0..5]              maps to "MDT", "CSC", "RPC", "TGC", "STGC", "MM"
00050 // ==============================================================================
00051 //
00052 // Inheritance
00053 // -----------
00054 // Inherits from DetectorDescription/AtlasDetectorID.
00055 // It is inherited by the following technology-specific concrete classes.
00056 //
00057 //   MdtIdHelper factory of MDT-specific Identifiers and Ranges
00058 //   CscIdHelper factory of CSC-specific Identifiers and Ranges
00059 //   RpcIdHelper factory of RPC-specific Identifiers and Ranges
00060 //   TgcIdHelper factory of TGC-specific Identifiers and Ranges
00061 
00062 // for nSW
00063 //   sTgcIdHelper factory of sTGC-specific Identifiers and Ranges
00064 //   MmIdHelper factory of MicroMegas-specific Identifiers and Ranges
00065 
00066 //
00067 // Methods for External Use
00068 // ------------------------
00069 // int stationNameIndex(const std::string name)   (returns stationName index)
00070 // int technologyIndex(const std::string name)    (returns technology index)
00071 // std::string stationNameString(const int index) (returns stationName string)
00072 // std::string technologyString(const int index)  (returns technology string)
00073 //
00074 // Author
00075 // ------
00076 // Steven Goldfarb <Steven.Goldfarb@cern.ch>
00077 //
00078 // Compact Id implementation by
00079 // Ketevi A. Assamagan <ketevi@bnl.gov>
00080 // CERN, February 20th, 2003
00081 //
00082 
00083 // for nSW
00084 // Nektarios Chr. Benekos <nectarios.benekos@cern.ch>
00085 // CERN, Feb. 2013
00086 // ******************************************************************************
