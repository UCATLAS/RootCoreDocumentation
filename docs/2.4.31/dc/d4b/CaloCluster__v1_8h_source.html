679  inline unsigned CaloCluster_v1::sampVarIdx(const CaloCluster_v1::CaloSample s) const {
680  const uint32_t& pattern= m_samplingPattern;
681  //std::cout << "Pattern=" << std::hex << pattern << std::dec << ", Sampling=" << s << std::endl;
682  if ((pattern & (0x1U << s))==0)
683  return CaloSampling::Unknown;
684  else {
685  if (s==0) return 0; //shifting a 32-bit int by 32 bits is undefined behavior!
686  return __builtin_popcount(pattern << (32-s));
687  //Explanation: Need to get the number of bit (=samples) before the sampling in question
688  //Shift to the left, so bits after the sampling in question fall off the 32bit integer
689  //Then use gcc builtin popcount to count the numbers of 1 in the rest
690  //AFAIK, this builtin is available for gcc, icc and clang (as well on ARM)
691  }
692  }
