163  bool isInDeltaR( const xAOD::IParticle& p1, const xAOD::IParticle& p2,
164  double dR, bool useRapidity=true )
165  {
166  const double dPhi = std::abs( xAOD::P4Helpers::deltaPhi(p1,p2) ); // in [-pi,pi)
167  if ( CxxUtils::fpcompare::greater(dPhi,dR) ) return false; // <==
168  if (useRapidity) {
169  const double dRapidity = std::abs( xAOD::P4Helpers::deltaRapidity(p1,p2) );
170  if ( CxxUtils::fpcompare::greater(dRapidity,dR) ) return false; // <==
171  const double deltaR2 = dRapidity*dRapidity + dPhi*dPhi;
172  if ( CxxUtils::fpcompare::greater(deltaR2,dR*dR) ) return false; // <==
173  return true;

175  else {

179  if ( CxxUtils::fpcompare::greater(deltaR2,dR*dR) ) return false; // <==
