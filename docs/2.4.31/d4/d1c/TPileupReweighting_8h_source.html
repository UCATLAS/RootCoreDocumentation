350  struct CompositeTrigger {
351  int op;
352  CompositeTrigger* trig1;
353  CompositeTrigger* trig2;
354  TString val;
355  std::vector<TString> subTriggers; //only set for top-level object
356  CompositeTrigger() : op(0),trig1(0),trig2(0),val("") { }
357  ~CompositeTrigger() { if(trig1) delete trig1; if(trig2) delete trig2; }
358  double eval(std::map<TString, std::map<Int_t, std::map<Int_t, Float_t> > >& m, int run, int lbn, const TPileupReweighting* tool) {
359  switch(op) {
360  case 0: if(m[val][run].find(lbn)==m[val][run].end() || !m[val][run][lbn] || !tool->passTriggerBeforePrescale(val)) return 0; /*trigger/failed disabled, so cannot contribute*/
361  return 1./m[val][run][lbn];
362  case 1: return 1. - (1. - trig1->eval(m,run,lbn,tool))*(1.-trig2->eval(m,run,lbn,tool)); //OR
363  case 2: return trig1->eval(m,run,lbn,tool)*trig2->eval(m,run,lbn,tool);
364  default: return 1;
365  }
366  }
367  void getTriggers(std::vector<TString>& s) {
368  if(trig1==0&&trig2==0&&val.Length()>0) s.push_back(val);
369  else { trig1->getTriggers(s); trig2->getTriggers(s); }
370  }

373  long out(0);

375  return out;
376  }
