8  // Here we define the trait classes. The basic problem is
9  // this:
10  //
11  // Some classes are attached to the navigation as basic objects
12  // for example - MuonFeature
13  // Other classes are attached to the navigation as containers of
14  // objects for example - egamma's are attached as egammaContainers
15  // Still other classes are not attached at all, but can be gotten
16  // from the ancestor method in the TDT, such as the L1 items
17  // We need to make this complexity invisible to the user, so they
18  // can just call whatever they want to match to and not worry
19  // about how it interacts with the navigation
20  // We solve this with traits classes (thanks to Scott Snyder for
21  // the suggestion) which tell us whether or not a class is a
22  // container and if it is, allows us to map from the class to
23  // the container class and similarly for ancest style objects
24 
25  struct DirectAttached {}; // identifies objects as attached directly
26  struct AncestorAttached {}; // identifies objects as attached via ancest
27 
28  // by default, we assume a class is attached directly
29  template<typename T>
30  struct ClassTraits {
31  typedef T type;
32  };

34 }
