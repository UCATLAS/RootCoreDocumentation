118 int PseudoJetGetter::
119 append(const TList& inputs, PseudoJetVector& psjs, const LabelIndex* pli) const {
120  int nskip = 0;
121  jet::IConstituentUserInfo::Index labidx = 0;
122  if ( pli != 0 ) labidx = pli->index(m_label);
123  else ATH_MSG_WARNING("Index-to-label map is not supplied.");
124  // if ( m_ghostscale ) labidx = -labidx;
125  if(!m_negEnergyAsGhosts && m_ghostscale) labidx = -labidx;
126  ATH_MSG_DEBUG( "Ghost scale = " << m_ghostscale << "; idx = " << labidx );
127 
129 #ifdef USE_BOOST_AUTO
130  for ( BOOST_AUTO(iinp, inputs.begin()); iinp!=inputs.end(); ++iinp ) {
131  BOOST_AUTO(ppar, *iinp); // IParticle pointer
132 #else
133  for ( auto iinp=inputs.begin(); iinp!=inputs.end(); ++iinp ) {
134  auto ppar = *iinp; // IParticle pointer
135 #endif
136  if ( ppar == 0 || (m_skipNegativeEnergy && ppar->e() <= 0.0) ) {
137  if ( ppar == 0 ) ATH_MSG_DEBUG("NUll object!");
138  else ATH_MSG_VERBOSE("Skipping cluster with E = " << ppar->e());
139  ++nskip;
140  continue;
141  }
142  // Take momentum from IParticle.
143  fastjet::PseudoJet psj(ppar->p4());
144  // Form EM scale, use uncalibrated four-vector.1
145  if ( m_emtopo ) {
146  const xAOD::CaloCluster* pclu = dynamic_cast<const xAOD::CaloCluster*>(ppar);
147  if ( pclu == 0 ) {
148  ATH_MSG_WARNING("EM particle is not type CaloCluster");
149  continue;
150  }
151  psj.reset(pclu->p4(xAOD::CaloCluster::UNCALIBRATED));
152  }
153 
154  if( m_negEnergyAsGhosts) {
155  if (ppar->e() <= 0.0) { // Important use the IParticle::e(), because IParticle::p4().e() might always be >0.
156  psj.reset_momentum(psj.px(), psj.py(), psj.pz(), std::abs( psj.e() )); // make sure e>0 in fastjet
157  psj *= m_ghostscale;
158  }
159  } else if ( m_ghostscale ) psj *= m_ghostscale;
160 
161  bool show = psjs.size() < 20;
162  if ( show ) ATH_MSG_VERBOSE("index/label: " << labidx << "/" << m_label);
163  if ( show ) ATH_MSG_VERBOSE("old/p4/new"
164  << " pt: " << ppar->pt() << "/" << ppar->p4().Pt() << "/" << psj.pt()
165  << ", pz: /" << ppar->p4().Pz() << "/" << psj.pz()
166  << ", p: /" << ppar->p4().P() << "/"
167  << ", E: " << ppar->e() << "/" << ppar->p4().E() << "/" << psj.e()
168  << ", m: " << ppar->m() << "/" << ppar->p4().M() << "/" << psj.m()
169  << ", eta: " << ppar->eta() << "/" << ppar->p4().Eta() << "/" << psj.eta()
170  );
171  if ( pli != 0 ) {
172  jet::IConstituentUserInfo* pcui = this->buildCUI(ppar, labidx, pli);
173  psj.set_user_info(pcui);
174  if ( show ) ATH_MSG_VERBOSE("User info particle: " << pcui->particle());
175  }
176  psjs.push_back(psj);
177  } // end loop over input
178 
179  if ( nskip ) ATH_MSG_DEBUG("Skipped constituent count: " << nskip);
180  ATH_MSG_DEBUG("After append, PseudoJet count is " << psjs.size());
181  return 0;
182 }
