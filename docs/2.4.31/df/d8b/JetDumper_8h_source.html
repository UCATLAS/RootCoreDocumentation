184 template<typename T>
185 int JetDumper::dump_object_after_prefix(const T* pjet) const {
186  const double mevtogev = 0.001;
187  const int wname = 30;
188  if ( pjet == 0 ) {
189  msg() << endmsg;
190  ATH_MSG_ERROR("Null jet.");
191  return 1;
192  }
193  // One line summary.
194  std::ostringstream ssjetline;
195  if ( m_linedetail > 0 ) {
196  ssjetline << "pT =" << std::setw(6) << std::fixed << std::setprecision(1) << mevtogev*pjet->pt() << " GeV";
197  ssjetline << ", m =" << std::setw(6) << std::fixed << std::setprecision(1) << mevtogev*pjet->m() << " GeV";
198  ssjetline << ", eta =" << std::setw(5) << std::fixed << std::setprecision(2) << pjet->eta();
199  ssjetline << ", phi =" << std::setw(5) << std::fixed << std::setprecision(2) << pjet->phi();
200  }
201  if ( m_linedetail > 1 ) extra_info(pjet, ssjetline, m_linedetail);
202  msg() << ssjetline.str() << endmsg;
203  // All properties.
204  if ( m_detail > 1 ) {
205  NameList fnames = m_fmoms;
206  NameList inames = m_imoms;
207  NameList bnames = m_bmoms;
208  NameList cbnames = m_cbmoms;
209  NameList snames = m_smoms;
210  NameList fvnames = m_fvmoms;
211  NameList elnames = m_elmoms;
212  NameList vinames = m_vimoms;
213  NameList vfnames = m_vfmoms;
214  NameList apnames = m_apmoms;
215  if ( fnames.size() == 0 ) fnames = get_moment_keys<T,float>(pjet);
216  ATH_MSG_INFO(" " << m_objtypename << " has " << fnames.size() << " float attributes:");
217  for ( NameList::const_iterator inam=fnames.begin(); inam!=fnames.end(); ++inam ) {
218  Name name = *inam;
219  float val;
220  get_moment(pjet, name, val);
221  if ( fabs(val) > m_floatmax ) {
222  ATH_MSG_INFO(std::setw(wname) << name << ":" << std::setw(16) << std::scientific << std::setprecision(3) << val);
223  } else {
224  ATH_MSG_INFO(std::setw(wname) << name << ":" << std::setw(12) << std::fixed << std::setprecision(3) << val);
225  }
226  }
227  if ( inames.size() == 0 ) inames = get_moment_keys<T,int>(pjet);
228  ATH_MSG_INFO(" " << m_objtypename << " has " << inames.size() << " int attributes:");
229  for ( NameList::const_iterator inam=inames.begin(); inam!=inames.end(); ++inam ) {
230  Name name = *inam;
231  int ival;
232  get_moment(pjet, name, ival);
233  ATH_MSG_INFO(std::setw(wname) << name << ":" << std::setw(12) << std::fixed << std::setprecision(3) << ival);
234  }
235  if ( bnames.size() == 0 ) bnames = get_moment_keys<T,int>(pjet);
236  ATH_MSG_INFO(" " << m_objtypename << " has " << bnames.size() << " bool attributes:");
237  for ( NameList::const_iterator inam=bnames.begin(); inam!=bnames.end(); ++inam ) {
238  Name name = *inam;
239  bool val;
240  get_moment(pjet, name, val);
241  std::string sval = val ? "true" : "false";
242  ATH_MSG_INFO(std::setw(wname) << name << ":" << std::setw(6) << sval);
243  }
244  if ( cbnames.size() == 0 ) cbnames = get_moment_keys<T,int>(pjet);
245  ATH_MSG_INFO(" " << m_objtypename << " has " << cbnames.size() << " cbool attributes:");
246  for ( NameList::const_iterator inam=cbnames.begin(); inam!=cbnames.end(); ++inam ) {
247  Name name = *inam;
248  char val;
249  get_moment(pjet, name, val);
250  std::string sval = val ? "true" : "false";
251  ATH_MSG_INFO(std::setw(wname) << name << ":" << std::setw(6) << sval);
252  }
253  ATH_MSG_INFO(" " << m_objtypename << " has " << snames.size() << " string attributes:");
254  for ( NameList::const_iterator inam=snames.begin(); inam!=snames.end(); ++inam ) {
255  Name name = *inam;
256  std::string sval;
257  get_moment(pjet, name, sval);
258  ATH_MSG_INFO(std::setw(wname) << name << ":" << std::setw(30) << sval);
259  }
260  ATH_MSG_INFO(" " << m_objtypename << " has " << vinames.size() << " vector<int> attributes:");
261  for ( NameList::const_iterator inam=vinames.begin(); inam!=vinames.end(); ++inam ) {
262  Name name = *inam;
263  std::vector<int> vals;
264  get_moment(pjet, name, vals);
265  if ( vals.size() == 0 ) {
266  ATH_MSG_INFO(std::setw(wname) << name << ": No entries.");
267  } else {
268  int widx = 1;
269  if ( vals.size() > 10 ) widx = 2;
270  for ( unsigned int ival=0; ival<vals.size(); ++ival ) {
271  ATH_MSG_INFO(std::setw(wname) << name << "[" << std::setw(widx) << ival << "]: "
272  << std::setw(8) << vals[ival]);
273  }
274  }
275  }
276  ATH_MSG_INFO(" " << m_objtypename << " has " << vfnames.size() << " vector<float> attributes:");
277  for ( NameList::const_iterator inam=vfnames.begin(); inam!=vfnames.end(); ++inam ) {
278  Name name = *inam;
279  std::vector<float> vals;
280  get_moment(pjet, name, vals);
281  if ( vals.size() == 0 ) {
282  ATH_MSG_INFO(std::setw(wname) << name << ": No entries.");
283  } else {
284  int widx = 1;
285  if ( vals.size() > 10 ) widx = 2;
286  for ( unsigned int ival=0; ival<vals.size(); ++ival ) {
287  ATH_MSG_INFO(std::setw(wname) << name << "[" << std::setw(widx) << ival << "]: "
288  << std::setw(12) << std::fixed << std::setprecision(3) << vals[ival]);
289  }
290  }
291  }
292  ATH_MSG_INFO(" " << m_objtypename << " has " << apnames.size() << " associated particle vectors:");
293  for ( NameList::const_iterator inam=apnames.begin(); inam!=apnames.end(); ++inam ) {
294  Name name = *inam;
295  if ( name.find("MuonSegment") != std::string::npos ) {
296  MSVector vals;
297  getAssociatedParticles(pjet, name, vals);
298  MSELVector els;
299  getAssociatedLinks(pjet, name, els);
300  std::string slab = "entries";
301  if ( vals.size() == 1 ) slab = "entry";
302  if ( vals.size() > 0 ) slab += ":";
303  ATH_MSG_INFO(std::setw(wname) << name << " has " << vals.size() << " " << slab);
304  if ( els.size() == vals.size() ) {
305 #ifdef USE_BOOST_FOREACH
306  BOOST_FOREACH(const MSEL& el, els) {
307  ElementLink<xAOD::MuonSegmentContainer> newel(el);
308 #else
309  for (const MSEL& el : els) {
310 #endif
311  ATH_MSG_INFO(std::setw(wname+2) << "" << el.dataID() << "[" << el.index() << "]");
312  }
313  } else {
314 #ifdef USE_BOOST_FOREACH
315  BOOST_FOREACH(const xAOD::MuonSegment* ppar, vals) {
316 #else
317  for (auto ppar : vals) {
318 #endif
319  ATH_MSG_INFO(std::setw(wname+2) << "" << ppar->container() << "[" << ppar->index() << "]");
320  }
321  }
322  } else {
323  APVector vals;
324  getAssociatedParticles(pjet, name, vals);
325  APELVector els;
326  getAssociatedLinks(pjet, name, els);
327  std::string slab = "entries";
328  if ( vals.size() == 1 ) slab = "entry";
329  if ( vals.size() > 0 ) slab += ":";
330  ATH_MSG_INFO(std::setw(wname) << name << " has " << vals.size() << " " << slab);
331  // Show container names if found, otherwise container addresses.
332  if ( els.size() == vals.size() ) {
333 #ifdef USE_BOOST_FOREACH
334  BOOST_FOREACH(const APEL& el, els) {
335 #else
336  for (const APEL& el : els) {
337 #endif
338  ATH_MSG_INFO(std::setw(wname+2) << "" << el.dataID() << "[" << el.index() << "]");
339  }
340  } else {
341 #ifdef USE_BOOST_FOREACH
342  BOOST_FOREACH(const xAOD::IParticle* ppar, vals) {
343 #else
344  for (auto ppar : vals) {
345 #endif
346  ATH_MSG_INFO(std::setw(wname+2) << "" << ppar->container() << "[" << ppar->index() << "]");
347  }
348  }
349  }
350  }
351  if ( fvnames.size() == 0 ) fvnames = get_moment_keys<T,float>(pjet);
352  ATH_MSG_INFO(" " << m_objtypename << " has " << fvnames.size() << " four-vector attributes (pT m eta phi):");
353  for ( NameList::const_iterator inam=fvnames.begin(); inam!=fvnames.end(); ++inam ) {
354  Name name = *inam;
355  FourVector val;
356  get_moment(pjet, name, val);
357  ATH_MSG_INFO(std::setw(wname) << name << ":"
358  << std::setw(12) << std::fixed << std::setprecision(3) << val.Pt()
359  << std::setw(12) << std::fixed << std::setprecision(3) << val.M()
360  << std::setw(12) << std::fixed << std::setprecision(3) << val.Eta()
361  << std::setw(12) << std::fixed << std::setprecision(3) << val.Phi()
362  );
363  }
364  ATH_MSG_INFO(" " << m_objtypename << " has " << elnames.size() << " element-link attributes:");
365  for ( NameList::const_iterator inam=elnames.begin(); inam!=elnames.end(); ++inam ) {
366  std::string name = *inam;
367  std::string sel = get_moment_as_string(pjet, name);
368  ATH_MSG_INFO(std::setw(wname) << name << ": " << sel);
369  }
370  }
371  return 0;

387 template<typename TList>
388 int JetDumper::dump_collection(const TList* pjets) const {
389  int njet = pjets->size();
390  ATH_MSG_INFO(" " << m_objtypename << " multiplicity: " << njet << " [" << m_cname << "]");
391  std::string line = "-------------------------------------------------------------------------";
392  int jstat = 0;
393  if ( m_detail > 0 ) {
394  unsigned int njetshow = njet;
395  if ( m_maxobj > 0 && m_maxobj < njet ) njetshow = m_maxobj;
396  typename TList::const_iterator itjet = pjets->begin();
397  typedef typename TList::const_iterator::reference TValue;
398  std::vector<std::string> labs;
399  unsigned int wlab = 0;
400  for ( unsigned int ijet=0; ijet<njetshow; ++ijet ) {
401  TValue pjet = *itjet++;
402  std::string lab = object_label(pjet, m_objtypename);
403  labs.push_back(lab);
404  if ( lab.size() > wlab ) wlab = lab.size();
405  }
406  itjet = pjets->begin();
407  int widx = 1;
408  if ( njetshow > 10 ) widx = 2;
409  if ( njetshow > 100 ) widx = 3;
410  if ( njetshow > 1000 ) widx = 4;
411  for ( unsigned int ijet=0; ijet<njetshow; ++ijet ) {
412  TValue pjet = *itjet++;
413  if ( m_detail > 1 && ijet==0 ) ATH_MSG_INFO(line);
414  // One line summary.
415  msg() << MSG::INFO << " ";
416  if ( m_prefix == 1 || m_prefix > 2 ) {
417  msg() << std::setw(wlab) << labs[ijet] << " ";
418  }
419  if ( m_prefix < 2 ) {
420  msg() << std::setw(widx) << ijet;
421  } else {
422  msg() << m_cname;
423  msg() << "[" << std::setw(widx) << ijet << "]";
424  }
425  msg() << ": ";
426  jstat += dump_object_after_prefix(pjet);
427  if ( m_detail > 1 ) ATH_MSG_INFO(line);
428  }
429  }
430  return jstat;
431 }
