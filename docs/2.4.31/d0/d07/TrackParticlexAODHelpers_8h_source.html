214  inline
215  double pTErr2Unsafe(const xAOD::TrackParticle *tp) {
216 
217  // / d \2 / d \2 d d
218  // pt_err^2 =| ------ (pt) *sigma_q/p | + | -------- pt * sigma_theta | + ----- pt ------- pt * sigma_theta_qp
219  // \ d q/p / \ d theta / d qp d theta
220 
221  // d / d pt 2 d pt \ / d pt \2
222  // = ------ (pt) * | ----- sigma_q/p + ------- * sigma_theta_qp | + | ------- * sigma_theta |
223  // d q/p \ d q/p d theta / \ d theta /
224 
225  // d pt
226  // ------ pt = - -------------
227  // d q/p fabs(qOverP)
228 
229  // d pt
230  // ------- pt = -----------
231  // d theta tan(theta)
232 
233  double pt = tp->pt();
234  double diff_qp = - pt / std::abs(tp->qOverP());
235 
236  double diff_theta = pt / tan( tp->theta() );
237 
238  // since 1/tan (thata) defined for all floats 0..M_PI, don't have to protect :
239  //
240  // double diff_theta = ( std::abs(tp->theta()-M_PI_2) > std::numeric_limits<float>::epsilon() * 2
241  // ? pt / tan( tp->theta() )
242  // : 0 )
243 
244 
245  const std::vector<float>&cov= tp->definingParametersCovMatrixVec();
246  // elements in definingParametersCovMatrixVec should be : 0: sigma_d0^2,
247  // 1: sigma_d0_z0, sigma_z0^2,
248  // 3: sigma_d0_phi, sigma_z0_phi, sigma_phi^2
249  // 6: sigma_d0_th, sigma_z0_th, sigma_phi_th, sigma_th^2
250  // 10: sigma_d0_qp, sigma_z0_qp, sigma_phi_qp, sigma_th_qp, sigma_qp^2
251 
252  double pt_err2 = diff_qp * (diff_qp * cov[14] + diff_theta * cov[13] ) + sqr(diff_theta) * cov[9];
253  return pt_err2;
254  }
