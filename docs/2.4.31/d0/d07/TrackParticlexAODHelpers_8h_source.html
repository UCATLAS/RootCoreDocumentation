00045       double sin_phi = sin(track_phi0);
00046       double cos_phi = cos(track_phi0);

00215     double pTErr2Unsafe(const xAOD::TrackParticle *tp) {
00216 
00217       //            /   d                       \2    /   d                        \2          d       d
00218       // pt_err^2 =|  ------   (pt) *sigma_q/p  |  + | --------   pt * sigma_theta |    +    ----- pt ------- pt * sigma_theta_qp
00219       //            \  d q/p                    /     \ d theta                    /          d qp    d theta
00220 
00221       //                d             / d pt           2   d pt                     \   / d pt                  \2
00222       //          =   ------   (pt) * | ----- sigma_q/p  + ------- * sigma_theta_qp | + | ------- * sigma_theta |
00223       //               d q/p          \ d q/p              d theta                  /   \ d theta               /
00224 
00225       //   d                pt
00226       // ------ pt  =  - -------------
00227       //  d q/p          fabs(qOverP)
00228 
00229       //   d                pt
00230       //  ------- pt =   -----------
00231       //  d theta         tan(theta)
00232 
00233       double pt = tp->pt();
00234       double diff_qp =  - pt / std::abs(tp->qOverP());
00235 
00236       double diff_theta = pt / tan( tp->theta() );
00237 
00238       // since 1/tan (thata) defined for all floats 0..M_PI, don't have to protect :
00239       //
00240       // double diff_theta = ( std::abs(tp->theta()-M_PI_2) > std::numeric_limits<float>::epsilon() * 2
00241       //                      ? pt / tan( tp->theta() )
00242       //                      : 0 )
00243 
00244       
00245       const std::vector<float>&cov= tp->definingParametersCovMatrixVec();
00246       // elements in definingParametersCovMatrixVec should be : 0: sigma_d0^2,
00247       //                                                        1: sigma_d0_z0,  sigma_z0^2,
00248       //                                                        3: sigma_d0_phi, sigma_z0_phi, sigma_phi^2
00249       //                                                        6: sigma_d0_th,  sigma_z0_th,  sigma_phi_th, sigma_th^2
00250       //                                                       10: sigma_d0_qp,  sigma_z0_qp,  sigma_phi_qp, sigma_th_qp, sigma_qp^2
00251 
00252       double pt_err2 = diff_qp * (diff_qp * cov[14] + diff_theta * cov[13] ) + sqr(diff_theta) * cov[9];
00253       return pt_err2;
00254     }
