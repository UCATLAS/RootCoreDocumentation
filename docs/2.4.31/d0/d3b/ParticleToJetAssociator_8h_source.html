172  ParticleToJetAssociator::associateParticlesToJets(jetcollection_t* theJets,
173  const coll* particleContainer,
174  const std::string& constituentName) {
175 
176  std::vector<ConstituentType*> vecTrackConst;
177  ATH_MSG_VERBOSE("Number of Jets : " << theJets->size());
178  if (theJets->size()<1) return vecTrackConst;
179  ATH_MSG_VERBOSE("Number of Particles of Type " << constituentName << " : " << particleContainer->size());
180 
181  for (jetcollection_t::iterator itr = theJets->begin(); itr != theJets->end(); ++itr) {
182  // NameType name(constituentName);
183  vecTrackConst.push_back(new ConstituentType);
184  }
185 
186  int index(0);
187  if(!m_shareTracks) {
188  // Case 1: associate a track to only one jet:
189  // loop through track particles and add to closest jet
190  typename coll::const_iterator tpItr = particleContainer->begin();
191  for (; tpItr!=particleContainer->end() ; tpItr++) {
192  double pt = (*tpItr)->pt();
193  double deltaRMatch(1000);
194  ATH_MSG_VERBOSE("Particle-Jet association (no sharing). Jet pT= " << pt);
195  bool findAMatch = false;
196  int i(0);
197  xAOD::Jet *theJet = 0;
198  for (jetcollection_t::iterator itr = theJets->begin(); itr != theJets->end(); ++itr) {
199  double r = (*itr)->p4().DeltaR((*tpItr)->p4());
200  if ( r < deltaRMatch ) {
201  deltaRMatch = r;
202  index = i;
203  findAMatch = true;
204  theJet = (*itr);
205  }
206  i++;
207  }
208 
209  ATH_MSG_VERBOSE(" matching: find= " << findAMatch << " dR=" << deltaRMatch);
210  double trackCone = m_trackCone;
211  if( m_useVariableSizedTrackCone ) {
212  if(0 != theJet) trackCone = getConeSize(theJet->pt());
213  ATH_MSG_VERBOSE("Using Variable Sized Cone of " << trackCone);
214  } else {
215  ATH_MSG_VERBOSE("Using Fixed Sized Cone of " << trackCone);
216  }
217  if (findAMatch && deltaRMatch < trackCone ) {
218  // FF: changed this to assume an STL container or at least push_back() functionality
219  // note that this does not have any mechanism for updating particle weights etc.
220  // vecTrackConst[index]->set_association(particleContainer, (*tpItr), 0.);
221  vecTrackConst[index]->push_back(*tpItr);
222  ATH_MSG_VERBOSE(" added constituent to Jet " << index);
223  }
224  }
225  } else {
226  // Case 2: can associate a track to many jets:
227  // loop on jets and tracks
228  jetcollection_t::iterator jetItr = theJets->begin();
229  jetcollection_t::iterator jetEnd = theJets->end();
230  for(; jetItr!=jetEnd; jetItr++) {
231  typename coll::const_iterator tpItr = particleContainer->begin();
232  typename coll::const_iterator tpEnd = particleContainer->end();
233  for(; tpItr!=tpEnd; tpItr++) {
234  double dR = (*jetItr)->p4().DeltaR((*tpItr)->p4());
235  double trackCone = m_trackCone;
236  if( m_useVariableSizedTrackCone ) {
237  trackCone = getConeSize((*jetItr)->pt());
238  ATH_MSG_VERBOSE("Using Variable Sized Cone of " << trackCone);
239  } else {
240  ATH_MSG_VERBOSE("Using Fixed Sized Cone of " << trackCone);
241  }
242  if(dR<trackCone) {
243  ATH_MSG_VERBOSE("Particle-jet association (with sharing): "
244  << " jet pt,phi,eta= " << (*jetItr)->pt() << " " << (*jetItr)->phi() << " " << (*jetItr)->eta()
245  << " trk pt,phi,eta= " << (*tpItr)->pt() << " " << (*tpItr)->phi() << " " << (*tpItr)->eta());
246  // FF: changed this to assume an STL container or at least push_back() functionality
247  // note that this does not have any mechanism for updating particle weights etc.
248  // vecTrackConst[index]->set_association( particleContainer, (*tpItr), 0);
249  vecTrackConst[index]->push_back(*tpItr);
250  }
251  }
252  index++;
253  }
254  }

263  }
