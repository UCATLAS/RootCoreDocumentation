197 template <size_t index>
198 StatusCode InDet::ParamAccessor<index>::access( const xAOD::TrackParticle& track,
199  const xAOD::Vertex* vertex )
200 {
201  m_paramValue = track.definingParameters()[index];
202  if (index == 1) { // if this is a z-accessor (should be evaluated at compile-time)
203  if (vertex != nullptr) {
204  // if vertex is provided we need to cut w.r.t. the vertex
205  m_paramValue += track.vz() - vertex->z();
206  }
207  }
208  return StatusCode::SUCCESS;
209 }
