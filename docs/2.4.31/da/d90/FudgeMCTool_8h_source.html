111  static std::vector<float>* getEratio(std::vector<float> emaxs1, std::vector<float> Emax2)
112  {
113  std::vector<float> *eratio = new std::vector<float>();
114  for (unsigned int i = 0; i < emaxs1.size(); ++i)
115  eratio->push_back(emaxs1[i] + Emax2[i] == 0 ? 0 :
116  (emaxs1[i] - Emax2[i])/(emaxs1[i] + Emax2[i]));
117  return eratio;
118  }
