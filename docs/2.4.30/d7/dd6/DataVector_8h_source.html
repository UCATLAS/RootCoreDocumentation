00563 // These are the intermediate classes from which @c DataVector derives
00564 // in the case of multiple or virtual derivation.  The actual
00565 // definitions are in the @c icc file.  They have the following
00566 // properties:
00567 //   - They derive from those of @c B1, @c B2, and @c B3 that are not
00568 //     @c NoBase.
00569 //   - They perform (static) checking to ensure that there's a unique
00570 //     base class in the hierarchy.
00571 //   - They make the following types available from the base classes:
00572 //     - @c PtrVector
00573 //     - @c BaseContainer
00574 //     - @c size_type
00575 //     - @c difference_type
00576 //     - @c allocator_type
00577 //   - @c has_virtual is defined as @c true.
00578 //   - An override of @c dvlinfo_v is defined, to prevent ambiguities.
00579 template <class B1,
00580           class B2=DataModel_detail::NoBase,
00581           class B3=DataModel_detail::NoBase> struct VirtBases;
