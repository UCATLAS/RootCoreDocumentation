00008     // Here we define the trait classes.  The basic problem is
00009     // this:
00010     //
00011     // Some classes are attached to the navigation as basic objects
00012     // for example - MuonFeature
00013     // Other classes are attached to the navigation as containers of
00014     // objects for example - egamma's are attached as egammaContainers
00015     // Still other classes are not attached at all, but can be gotten
00016     // from the ancestor method in the TDT, such as the L1 items
00017     // We need to make this complexity invisible to the user, so they
00018     // can just call whatever they want to match to and not worry
00019     // about how it interacts with the navigation
00020     // We solve this with traits classes (thanks to Scott Snyder for
00021     // the suggestion) which tell us whether or not a class is a
00022     // container and if it is, allows us to map from the class to
00023     // the container class and similarly for ancest style objects
00024 
00025   struct DirectAttached {}; // identifies objects as attached directly
00026   struct AncestorAttached {}; // identifies objects as attached via ancest
00027 
00028     // by default, we assume a class is attached directly
00029     template<typename T>
00030     struct ClassTraits {
00031         typedef T type;
00032     };
