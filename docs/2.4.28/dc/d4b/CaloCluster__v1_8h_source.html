00679   inline unsigned CaloCluster_v1::sampVarIdx(const CaloCluster_v1::CaloSample s) const {
00680     const uint32_t& pattern= m_samplingPattern;
00681     //std::cout << "Pattern=" << std::hex << pattern << std::dec << ", Sampling=" << s << std::endl;
00682     if ((pattern & (0x1U << s))==0)
00683       return CaloSampling::Unknown;
00684     else {
00685       if (s==0) return 0; //shifting a 32-bit int by 32 bits is undefined behavior! 
00686       return __builtin_popcount(pattern << (32-s)); 
00687       //Explanation: Need to get the number of bit (=samples) before the sampling in question
00688       //Shift to the left, so bits after the sampling in question fall off the 32bit integer
00689       //Then use gcc builtin popcount to count the numbers of 1 in the rest
00690       //AFAIK, this builtin is available for gcc, icc and clang (as well on ARM)
00691     }
00692   }
